<script>
let users = JSON.parse(localStorage.getItem("users")) || {};

function showPage(page){
document.querySelectorAll('.section').forEach(s=>s.classList.remove('active'));
document.getElementById(page).classList.add('active');
}

function registerUser(){
let u=document.getElementById('regUser').value;
let p=document.getElementById('regPass').value;
if(users[u]) alert("User exists");
else{
users[u]=p;
localStorage.setItem("users",JSON.stringify(users));
alert("Registered Successfully");
showPage('login');
}
return false;
}

function loginUser(){
let u=document.getElementById('username').value;
let p=document.getElementById('password').value;
if(users[u] && users[u]===p){
alert("Login Success");
showPage('courses');
}else alert("Invalid Login");
return false;
}

function openCourse(course){
showPage('videos');
let content="";

if(course==='quant'){
content=`<iframe width="100%" height="300" src="https://www.youtube.com/embed/1QXODo6m2z0"></iframe>`;
}
else if(course==='reason'){
content=`<iframe width="100%" height="300" src="https://www.youtube.com/embed/2Vv-BfVoq4g"></iframe>`;
}
else{
content=`<iframe width="100%" height="300" src="https://www.youtube.com/embed/Zi-XLOBDo_Y"></iframe>`;
}

document.getElementById('videoContent').innerHTML=content;
}
</script>
